DayReecords& operator << (DayRecords& dr, const int del_index){
    dr.transactions[del_index - 1] = NULL;
    dr.ReduceTransDynamic();
}

void ReduceTransDynamic(){
    Entry *temp = new Entry[--size];
    int count = 0;
    for(int i = 0; i < size + 1; i++){
        if (transactions[i] != NULL){
            temp[count] = transactions[i]
            ++count;
        }
        else    
            continue;
    }
    delete [] transactions;
    transactions = temp;
}
